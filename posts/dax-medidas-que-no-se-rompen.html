Manos a la obra

import polars as pl

ventas = pl.read_parquet("ventas.parquet")
resumen = (
    ventas
    .group_by("categoria")
    .agg(pl.col("monto").sum().alias("total"))
    .sort("total", descending=True)
)
print(resumen.head())
